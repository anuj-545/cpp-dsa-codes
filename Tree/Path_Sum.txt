/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    void solve(TreeNode* root , int &targetSum , vector<int> &ds , int sum)// agar function me peeche jaana hai aur sum bhi tumhe peeche waala chaiyea to ye '&' mat lagao please wrna wo pichle sum ko yaad rakhega.
    {
        if(!root)return;//agar maan lo tree hi null hua to 

        sum = sum + root->val;

        if(root->left==NULL && root->right==NULL)
        {
            ds.push_back(sum);
            return ;
        }

        solve(root->left,targetSum,ds,sum);
        solve(root->right,targetSum,ds,sum);//ye tum soch rahe hoge ki sum to reupdate ho hi nahi rha previous pe to aisa nahi hai jaise hi previous  function pe jayega to wo previous sum hi lega .

    }
    bool hasPathSum(TreeNode* root, int targetSum) {
        vector<int> ds;
        int sum=0 ; 

        solve(root,targetSum,ds,sum);
        bool ans = false;

        for(int i=0 ; i<ds.size() ; i++)
        {
            if(targetSum == ds[i])
            {
                ans = true;
            }
            // cout<<ds[i]<<" ";
        }

        return ans;
    }
};
